    Complete Datalayer API interface. Provides all methods for interacting with the Datalayer platform.

            { data?: Record<string, unknown>; error?: string; success: boolean },
        >;
        getRuntimeDetails: (
            runtimeId: string,
        ) => Promise<{ data?: RuntimeData; error?: string; success: boolean }>;
        getSpaceItems: (
            spaceId: string,
        ) => Promise<
            { data?: Record<string, unknown>[]; error?: string; success: boolean },
        >;
        getUserSpaces: () => Promise<
            { error?: string; spaces?: SpaceData[]; success: boolean },
        >;
        isRuntimeActive: (
            podName: string,

                success: boolean;
            },
        >;
        listNotebooks: () => Promise<
            { data?: NotebookData[]; error?: string; success: boolean },
        >;
        listUserRuntimes: () => Promise<
            { data?: RuntimeData[]; error?: string; success: boolean },
        >;
        login: (

        ) => Promise<
            {
